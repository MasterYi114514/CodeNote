# 代码

## 思路一：通过队列维护时间窗口

```cpp
#include <iostream>
#include <queue>

using namespace std;

typedef pair<int, int> p;   // first: 时间（ti）, second: 国家

const int N = 1e5 + 10;
const int delta_time = 86400;

int n;
queue<p> q;

int cnt[N];             // cnt[i] 表示国家 i 有多少人
int num = 0;            // 表示当前有人的国家数

int main()
{
    cin >> n;
    for(int i = 0, t, k; i < n; ++i)
    {
        cin >> t >> k;
        for(int j = 0, tmp; j < k; ++j)
        {
            cin >> tmp;
            q.push({t, tmp});

            // cnt[tmp]++ == 0 表示 cnt[tmp] 之前为 0，现在有人了，所以国家数 num 加 1
            if(cnt[tmp]++ == 0) num++;
        }

        // 限制队列时间间隔在 delta_time 内
        while(q.size() && q.front().first + delta_time <= t)
        {
            // 执行出队逻辑
            auto front = q.front();
            q.pop();

            if(--cnt[front.second] == 0) num--;
        }

        cout << num << endl;
    }

    return 0;
}
```

## 思路二：通过滑动窗口进行维护

```cpp
#include <iostream>

using namespace std;

typedef pair<int, int> p;   // first: 时间（ti）, second: 国家

const int N = 1e5 + 10;
const int K = 3e5 + 10;
const int delta_time = 86400;

int n;

int front, rear;
p q[K];               // 队列，存储 {时间, 国家}

int cnt[N];             // cnt[i] 表示国家 i 有多少人
int num = 0;            // 表示当前有人的国家数

int main()
{
    cin >> n;
    for(int i = 0, t, k; i < n; ++i)
    {
        cin >> t >> k;
        for(int j = 0, tmp; j < k; ++j)
        {
            cin >> tmp;
            q[rear++] = {t, tmp};       // 执行入队逻辑

            // cnt[tmp]++ == 0 表示 cnt[tmp] 之前为 0，现在有人了，所以国家数 num 加 1
            if(cnt[tmp]++ == 0) num++;
        }

        // 限制队列时间间隔在 delta_time 内
        while(front < rear && q[front].first + delta_time <= t)
        {
            // 执行出队逻辑
            auto front_e = q[front++];

            if(--cnt[front_e.second] == 0) num--;
        }

        cout << num << endl;
    }

    return 0;
}
```
