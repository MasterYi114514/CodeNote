# 代码

此题的 N 只有 `1000` ，即使用遍历查找的时间复杂度为 $O(N^2)$ 也是够用的：

```cpp
#include <iostream>

using namespace std;

const int M = 110, N = 1010;

int m, n;
int q[N], front = 0, rear = -1;

int ret = 0;

void push(int x)
{
    if(rear - front + 1 == m) ++front;
    q[++rear] = x;
}

bool query(int x)
{
    for(int i = front; i <= rear; ++i)
    {
        if(q[i] == x) return true;
    }

    return false;
}

int main()
{
    cin >> m >> n;

    while(n--)
    {
        int x;
        cin >> x;
        if(query(x)) continue;

        push(x);
        ret++;
    }

    cout << ret << endl;

    return 0;
}
```

如果想进一步优化时间复杂度，可借用 `mp` 数组标记队列中的元素，只需 $O(1)$ 的时间完成查询，代码的总时间复杂度为 $O(N)$：

```cpp
#include <iostream>

using namespace std;

const int M = 110, N = 1010;

int m, n;
int q[N], front = 0, rear = -1;
bool mp[N];

int ret = 0;

void push(int x)
{
    if(rear - front + 1 == m) mp[q[front++]] = false;

    q[++rear] = x;
    mp[x] = true;
}

int main()
{
    cin >> m >> n;

    while(n--)
    {
        int x;
        cin >> x;
        if(mp[x]) continue;

        push(x);
        ret++;
    }

    cout << ret << endl;

    return 0;
}
```
