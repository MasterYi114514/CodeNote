# 代码

```cpp
#include <iostream>
#include <stack>
#include <string>

using namespace std;

int to_num(string num_s)
{
    int ret = 0;
    for (int i = 0; i < num_s.size(); i++)
    {
        ret = ret * 10 + (num_s[i] - '0');
    }
    return ret;
}

int main()
{
    stack<int> stk;
    char ch;
    string num_s;
    while(cin >> ch)
    {
        if(ch <= '9' && ch >= '0')
        {
            num_s += ch;
        }
        else if(ch == '.')
        {
            stk.push(to_num(num_s));
            num_s = "";
        }
        else if(ch == '@')
        {
            cout << stk.top() << endl;
            break;
        }
        else    // + - * /
        {
            int r = stk.top();
            stk.pop();
            int l = stk.top();
            stk.pop();

            if(ch == '+') stk.push(l + r);
            else if(ch == '-') stk.push(l - r);
            else if(ch == '*') stk.push(l * r);
            else if(ch == '/') stk.push(l / r);
        }
    }

    return 0;
}
```
