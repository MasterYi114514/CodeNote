# 代码

```cpp
#include <iostream>
#include <string>

using namespace std;

const int N = 110;

string s;
string ret;

int top = 0;    // 栈顶指针，有效下标为 [1, top]
int stk[N];     // 存储左括号的下标

bool is_matched[N];    // 标记每个下标对应的括号是否匹配

int main()
{
    cin >> s;
    for(int i = 0; i < s.size(); i++)
    {
        char ch = s[i];
        if(ch == '(' || ch == '[') stk[++top] = i;
        else // if(ch == ')' || ch == ']')
        {
            if(top == 0) continue;

            char left = s[stk[top]];
            if((left == '(' && ch == ')') || (left == '[' && ch == ']'))
            {
                is_matched[stk[top]] = true;
                is_matched[i] = true;
                top--;
            }
        }
    }

    // 输出匹配的括号
    for(int i = 0; i < s.size(); i++)
    {
        if(is_matched[i]) ret += s[i];
        else
        {
            if(s[i] == '(' || s[i] == ')') ret += "()";
            else if(s[i] == '[' || s[i] == ']') ret += "[]";
        }
    }

    cout << ret << endl;

    return 0;
}
```
