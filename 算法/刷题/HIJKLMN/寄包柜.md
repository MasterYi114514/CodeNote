# 代码

如果这里开二维静态数组 `a[N][N]` 会导致内存超限，因此使用 vector 来动态管理每个寄包柜的存储空间。

```cpp
#include <iostream>
#include <vector>

using namespace std;

const int N = 1e5 + 10;

int n, q;
vector<int> a[N];

int main()
{
    cin >> n >> q;
    while(q--)
    {
        int op;
        cin >> op;
        if(op == 1)
        {
            int i, j, k;
            cin >> i >> j >> k;
            if(a[i].size() <= j)
                a[i].resize(j + 1);

            a[i][j] = k;
        }
        else if(op == 2)
        {
            int i, j;
            cin >> i >> j;
            cout << a[i][j] << endl;
        }
    }

    return 0;
}
```
