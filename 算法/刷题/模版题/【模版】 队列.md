# 代码

```cpp
#include <iostream>

using namespace std;

const int N = 1e4 + 10;

int front = 0, rear = -1;
int queue[N];

void push(int x)
{
    queue[++rear] = x;
}

void pop()
{
    if(rear >= front) ++front;
    else cout << "ERR_CANNOT_POP" << endl;
}

void query()
{
    if(rear >= front) cout << queue[front] << endl;
    else cout << "ERR_CANNOT_QUERY" << endl;
}

void size()
{
    cout << rear - front + 1 << endl;
}

int main()
{
    int n;
    cin >> n;
    while(n--)
    {
        int op;
        cin >> op;

        if(op == 1)
        {
            int x;
            cin >> x;
            push(x);
        }
        else if(op == 2) pop();
        else if(op == 3) query();
        else if(op == 4) size();
    }
    
    return 0;
}
```
