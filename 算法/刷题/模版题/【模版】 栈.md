# 代码

```cpp
#include <iostream>
#include <string>

using namespace std;

typedef unsigned long long ull;

const int N = 1e6 + 10;

int top = 0;    // 栈顶指针
ull stack[N];

int T, n;

void push(ull x)
{
    stack[++top] = x;   // 有效下标从1开始
}

void pop()
{
    if(top > 0) top--;
    else cout << "Empty" << endl;
}

void query()
{
    if(top > 0) cout << stack[top] << endl;
    else cout << "Anguei!" << endl;
}

void size()
{
    cout << top << endl;
}

int main()
{
    cin >> T;
    while(T--)
    {
        cin >> n;
        top = 0;    // 每组测试数据开始前清空栈

        string op;
        for(int i = 1; i <= n; i++)
        {
            cin >> op;
            if(op == "push") 
            {
                ull x;
                cin >> x;
                push(x);
            }
            else if(op == "pop") pop();
            else if(op == "query") query();
            else if(op == "size") size();
        }
    }
}
```
