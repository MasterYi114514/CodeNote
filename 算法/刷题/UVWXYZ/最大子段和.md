# 代码

```cpp
#include <iostream>

using namespace std;

const int N = 2e5 + 10;
const int INF = 0x3f3f3f3f;

int n;
int sum[N] = { 0 };

int main()
{
    cin >> n;

    int pre_min = 0, ret = -INF;
    for(int i = 1; i <= n; i++)
    {
        cin >> sum[i];
        sum[i] += sum[i - 1];

        ret = std::max(ret, sum[i] - pre_min);

        pre_min = std::min(pre_min, sum[i]);
    }

    cout << ret << endl;
    return 0;
}
```
