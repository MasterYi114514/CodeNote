# 代码

```cpp
#include <iostream>

using namespace std;

const int N = 1e6 + 10;

int pushed[N], popped[N];

int q, n;

int stk[N];
int top, cur;       // cur指向popped数组中当前要匹配的元素
bool solve()
{
    for(int i = 1; i <= n; i++)
    {
        stk[++top] = pushed[i];
        while(top > 0 && cur <= n && stk[top] == popped[cur])
        {
            top--, cur++;
        }
    }

    return top == 0;
}

int main()
{
    cin >> q;
    while(q--)
    {
        cin >> n;
        top = 0, cur = 1;
        for(int i = 1; i <= n; i++)
        {
            cin >> pushed[i];
        }

        for(int i = 1; i <= n; i++)
        {
            cin >> popped[i];
        }

        if(solve()) cout << "Yes" << endl;
        else cout << "No" << endl;
    }

    return 0;
}
```
