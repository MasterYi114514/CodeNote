# 代码

```cpp
#include <iostream>

using namespace std;

const int N = 200;      // 确保比字符的 ASCII 码范围大即可

int n;
char root;
char lc[N], rc[N];

void dfs(char u)
{
    if(u == '*') return;    // 访问到空节点，返回

    cout << u;      // 访问当前节点
    dfs(lc[u]);     // 访问左子树
    dfs(rc[u]);     // 访问右子树
}

int main()
{
    cin >> n;

    // 处理根节点
    cin >> root;
    cin >> lc[root] >> rc[root];

    for(int i = 2; i <= n; i++)
    {
        char node;
        cin >> node;

        // 这里直接将 char 类型的字符设置为索引，实际上是用字符的 ASCII 码作为索引
        cin >> lc[node] >> rc[node];
    }

    // 进行前序遍历
    dfs(root);

    return 0;
}
```
