# 代码

## 方法一：三区间扩容

```cpp
class Solution 
{
public:
    void sortColors(vector<int>& nums) 
    {
        int l = 0, r = 0;
        for(int cur = 0; cur < nums.size(); cur++)
        {
            /*
                进入循环时，[0, cur-1] 已经处理完毕
                - [0, l-1] 全是 0
                - [l, r-1] 全是 1
                - [r, cur-1] 全是 2
            */
            if(nums[cur] == 0)
            {
                if(r < cur) swap(nums[l], nums[r]);
                swap(nums[l], nums[cur]);
                l++,r++;
            }
            else if(nums[cur] == 1)
            {
                swap(nums[r], nums[cur]);
                r++;
            }
            // else if(nums[cur] == 2)
            // 此时不需要操作          
        }
    }
};
```

## 方法二：双指针

```cpp
class Solution 
{
public:
    void sortColors(vector<int>& nums) 
    {
        int l = 0, r = nums.size();

        int i = 0;
        while(i < r)
        {
            /*
                被分为四个区间
                - [0, l-1] 全是 0
                - [l, i-1] 全是 1
                - [i, r-1] 未处理
                - [r, nums.size()-1] 全是 2
            */
            if(nums[i] == 0)
            {
                swap(nums[l], nums[i]);
                l++, i++;
            }
            else if(nums[i] == 1) i++;
            else if(nums[i] == 2)
            {
                swap(nums[i], nums[r-1]);
                r--;
            }
        }
    }
};
```
