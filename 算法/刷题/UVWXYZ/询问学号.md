# 代码

```cpp
#include <iostream>

using namespace std;

const int N = 2e6 + 10;

int n, m;
int a[N];

int main()
{
    cin >> n >> m;
    for(int i = 1; i <= n; i++)
    {
        cin >> a[i];
    }

    while(m--)
    {
        int pos;
        cin >> pos;
        cout << a[pos] << endl;
    }
}
```
