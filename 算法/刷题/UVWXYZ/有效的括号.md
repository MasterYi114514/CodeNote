# 代码

```cpp
class Solution {
public:
    bool isValid(string s) 
    {
        vector<char> stk;
        for(auto ch : s)
        {
            if(ch == ')')
            {
                if(stk.empty() || stk.back() != '(') return false;
                else stk.pop_back();
            }
            else if(ch == ']')
            {
                if(stk.empty() || stk.back() != '[') return false;
                else stk.pop_back();
            }
            else if(ch == '}')
            {
                if(stk.empty() || stk.back() != '{') return false;
                else stk.pop_back();
            }
            else stk.push_back(ch);
        }

        return stk.size() == 0;
    }
};
```
