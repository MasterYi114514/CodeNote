# 代码

```cpp
#include <iostream>

using namespace std;

const int N = 110;

int n, m;
int ne[N];

int main()
{
    cin >> n >> m;
    ne[0] = 1;
    for(int i = 1; i <= n; i++)
    {
        ne[i] = i % n + 1;
    }

    int t = 0;
    for(int i = 1; i <= n; i++)
    {
        for(int j = 1; j < m; j++)
        {
            t = ne[t];
        }

        cout << ne[t] << " ";
        ne[t] = ne[ne[t]];
    }

    return 0;
}
```
