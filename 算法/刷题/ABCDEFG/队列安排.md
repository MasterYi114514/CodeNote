# 代码

```cpp
#include <iostream>

using namespace std;

const int N = 1e5 + 10;

/*
    编号范围为 [1, n]
    选取 0 作为哨兵位
*/
int n, m;
int ne[N], pre[N];
bool removed[N];

int main()
{
    cin >> n;
    ne[0] = 1, pre[1] = 0;

    int k, p;
    for(int i = 2; i <= n; i++)
    {
        cin >> k >> p;
        if(p == 0)
        {
            // i 插入 k 的左边
            ne[i] = k;
            pre[i] = pre[k];

            pre[k] = i;
            ne[pre[i]] = i;
        }
        else if(p == 1)
        {
            // i 插入 k 的右边
            pre[i] = k;
            ne[i] = ne[k];

            ne[k] = i;
            pre[ne[i]] = i;
        }
    }

    cin >> m;
    while(m--)
    {
        int x;
        cin >> x;
        if(removed[x]) continue;

        removed[x] = true;
        ne[pre[x]] = ne[x];
        pre[ne[x]] = pre[x];
    }

    for(int i = ne[0]; i != 0; i = ne[i])
    {
        cout << i << ' ';
    }

    return 0;
}
```
