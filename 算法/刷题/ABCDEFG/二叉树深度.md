# 代码

```cpp
#include <iostream>
#include <algorithm>        // for std::max

using namespace std;

const int N = 1e6 + 10;

int n;
int lc[N], rc[N];

int ret = 0;        // 用于记录二叉树的最大深度
void dfs(int u, int depth)
{
    if(u == 0) return;

    ret = max(ret, depth);      // 更新最大深度
    dfs(lc[u], depth + 1);      // 访问左子树
    dfs(rc[u], depth + 1);      // 访问右子树
}

int main()
{
    cin >> n;
    for(int i = 1; i <= n; i++)
    {
        cin >> lc[i] >> rc[i];
    }

    dfs(1, 1);     // 从根节点开始，初始深度为1

    cout << ret << endl;     // 输出二叉树的最大深度
    
    return 0;
}
```
