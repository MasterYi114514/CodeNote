# 代码

```cpp
#include <iostream>

using namespace std;

const int N = 1e6 + 10;

int q;
int ne[N];

// 将 y 插入到 x 的后面
void insert(int x, int y)
{
    ne[y] = ne[x];
    ne[x] = y;
}

void find(int x)
{
    cout << ne[x] << endl;
}

void remove(int x)
{
    ne[x] = ne[ne[x]];
}

int main()
{
    cin >> q;

    while(q--)
    {
        int op;
        cin >> op;
        if(op == 1)
        {
            int x, y;
            cin >> x >> y;
            insert(x, y);
        }
        else if(op == 2)
        {
            int x;
            cin >> x;
            find(x);
        }
        else
        {
            int x;
            cin >> x;
            remove(x);
        }
    }

    return 0;
}
```
