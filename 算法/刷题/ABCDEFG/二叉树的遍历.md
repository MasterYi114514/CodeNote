# 代码

```cpp
#include <iostream>
#include <cstring>  // for memset

using namespace std;

const int N = 1e6 + 10;

int n;
int lc[N], rc[N];

bool visited[N];        // 记录节点是否被访问过

// 前序遍历
void dfs_f(int u)
{
    if(u == 0) return;

    cout << u << ' ';     // 访问当前节点
    visited[u] = true;   // 标记当前节点为已访问

    dfs_f(lc[u]);        // 访问左子树
    dfs_f(rc[u]);        // 访问右子树
}

// 中序遍历
void dfs_m(int u)
{
    if(u == 0) return;

    dfs_m(lc[u]);        // 访问左子树

    cout << u << ' ';     // 访问当前节点
    visited[u] = true;   // 标记当前节点为已访问

    dfs_m(rc[u]);        // 访问右子树
}


// 后序遍历
void dfs_l(int u)
{
    if(u == 0) return;

    dfs_l(lc[u]);        // 访问左子树
    dfs_l(rc[u]);        // 访问右子树

    cout << u << ' ';     // 访问当前节点
    visited[u] = true;   // 标记当前节点为已访问
}

int main()
{
    cin >> n;

    for(int i = 1; i <= n; i++)
    {
        cin >> lc[i] >> rc[i];
    }

    // 堆内存会自动初始化为 false
    // memset(visited, false, sizeof visited);
    dfs_f(1);    // 前序遍历
    std::cout << std::endl; // 输出换行

    memset(visited, false, sizeof visited);     // 重置访问数组
    dfs_m(1);    // 中序遍历
    std::cout << std::endl; // 输出换行

    memset(visited, false, sizeof visited);     // 重置访问数组
    dfs_l(1);    // 后序遍历
    std::cout << std::endl; // 输出换行

    return 0;
}
```
