# 代码

## 1. 枚举递归实现

```cpp
#include <iostream>
#include <string>

using namespace std;

string a, b;        // 分别用于存储中序与后序遍历
int idx[200];       // 后序遍历中每个字符的位置

void dfs(int l, int r)
{
    if(l > r) return;     // 递归出口：当区间无效时返回

    // 先选择可能的根节点
    int root = l;
    for(int i = l + 1; i <= r; ++i)
    {
        if(idx[a[i]] > idx[a[root]])
        {
            root = i;
        }
    }

    cout << a[root];       // 根
    dfs(l, root - 1);      // 左子树
    dfs(root + 1, r);      // 右子树
}

int main()
{
    cin >> a >> b;

    for(int i = 0; i < b.size(); i++)
    {
        idx[b[i]] = i;      // 记录后序遍历中每个字符的位置
    }

    dfs(0, a.size() - 1);     // 从整个区间开始递归

    return 0;
}
```

算法分析：
- 时间复杂度：O(n^2)，其中n是树的节点数。每次递归需要遍历当前区间来找到根节点。
- 空间复杂度：O(n)，主要用于存储输入字符串和递归调用栈。

## 2. 中序与后序遍历性质的运用

```cpp
#include <iostream>
#include <string>

using namespace std;

string a, b;        // 分别用于存储中序与后序遍历

void dfs(int l1, int r1, int l2, int r2)
{
    // 递归出口
    if(l1 > r1 || l2 > r2) return;

    cout << b[r2];          // 由于后序遍历的特点，其最后一个元素一定是当前子树的根节点

    int k = l1;             // 在中序遍历中找到根节点的位置
    while(a[k] != b[r2]) k++;

    // 递归访问左子树
    dfs(l1, k - 1, l2, l2 + (k - 1 - l1));     // 左子树的长度为 k - 1 - l1 + 1 = k - l1

    // 递归访问右子树
    dfs(k + 1, r1, l2 + (k - l1), r2 - 1);     // 右子树的长度为 r1 - (k + 1) + 1 = r1 - k
}

int main()
{
    cin >> a >> b;

    dfs(0, a.size() - 1, 0, b.size() - 1);     // 从整个序列开始递归

    return 0;
}
```

算法分析：
- 时间复杂度：O(n)，每个节点被访问一次。
- 空间复杂度：O(n)，主要用于存储输入字符串和递归调用栈。
