# 代码

```cpp
#include <iostream>

using namespace std;

const int N = 1e6 + 10;

int n;
int ne[N];

int h;

int main()
{
    cin >> n;

    for(int i = 1; i <= n; i++)
    {
        cin >> ne[i];
    }

    cin >> h;
    for(int cur = h; cur != 0; cur = ne[cur])
    {
        cout << cur << " ";
    }

    return 0;
}
```
