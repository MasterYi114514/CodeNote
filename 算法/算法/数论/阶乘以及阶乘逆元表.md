
当需要多次查询阶乘以及阶乘逆元表时，可以预处理出阶乘表和阶乘逆元表。

可以通过以下方法用 O(n) 的时间复杂度预处理出阶乘表和阶乘逆元表：

```cpp
const int N = 1e6 + 10;         // 根据需要调整大小
const int p = 1e9 + 7;          // 模数

using ll = long long;

int n;
ll fact[N];       // 阶乘表
ll inv_fact[N];   // 阶乘逆元表

// 计算 a^b mod p 的快速幂函数
ll qpow(ll a, ll b, ll mod)
{
    ll res = 1;
    a %= mod;
    while(b > 0)
    {
        // 当 b & 1 == 1 时，说明当前位是 1，需要将 a 的当前幂次乘到结果中
        // 当 b & 1 == 0 时，说明当前位是 0，不需要将 a 的当前幂次乘到结果中
        if(b & 1) res = res * a % mod;
        a = a * a % mod;
        b >>= 1;
    }
    return res;
}

// 整个预处理时间复杂度为 O(n)
void precompute()
{
    // 预处理获得阶乘表
    fact[0] = 1;
    for(int i = 1; i <= n; i++)
    {
        fact[i] = fact[i - 1] * i % p;
    }

    // 预处理获得阶乘逆元表
    inv_fact[n] = qpow(fact[n], p-2, p);    // 使用快速幂计算阶乘的逆元
    for(int i = n - 1; i >= 0; i--)         // 方向从 n-1 到 0 递推计算阶乘逆元表
    {
        inv_fact[i] = inv_fact[i + 1] * (i + 1) % p;    
    }
}
```
